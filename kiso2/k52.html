    /*
** プログラム名: kadai52
** 動作: 共用体を使ったデータの切り出し（その２）
** 作者: 自分の名前、学籍番号
** 日付: 今日の日付
*/
#include <stdio.h>

int main(void)
{
    union bitfield {
	unsigned e:8;
	unsigned f:5;
    } a;

    a.e = 200;
    printf("a.e = %d, a.f = %d\n", a.e, a.f);

    a.f = 200;
    printf("a.e = %d, a.f = %d\n", a.e, a.f);

    return 0;
}

  3. kadai52 を実行してください。
