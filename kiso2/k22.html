    /*
** プログラム名: kadai22
** 動作: 数値の入力
** 作者: 自分の名前、学籍番号
** 日付: 今日の日付

#include <stdio.h>

int main(void)
{
    int x, y;

    printf("整数を２つ入力してください:");
    scanf("%d%d", x, y);
    printf("%d + %d = %d\n", x, y, x + y);

    return 0;
}
