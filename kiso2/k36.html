#include <stdio.h>

int main(void)
{
    int n, sum = 0;

    printf("整数を入れてください：");
    scanf("%d", &n);
    
start_loop:
    if (n > 0) {
        sum += n--;
        goto start_loop;
    }

    printf("合計は %d です\n", sum);
    return 0;
}
