** プログラム名: kadai10

#include <stdio.h>

void menseki(int takasa, int haba)
{
    printf("高さ %d、幅 %d の長方形の面積は %d\n",
        takasa, haba, takasa * haba);
}

int main(void)
{
    /* ここにプログラムを書く */
}
